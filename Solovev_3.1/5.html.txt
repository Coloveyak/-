function Student(name, age, phone) {
  this.name = name;
  this.age = age;
  this.phone = phone;
  this.getAveragePhone = function() {
    return this.phone.reduce((a, b) => a + b) / this.phone.length;
  };
}

const student1 = new Student("Alex", 17, 79055923949);
console.log(student1.name); 
console.log(student1.age); 
console.log(student1.phone); 
console.log(student1.getAveragePhone()); 
</script>

   // Операции в качестве значений
   add(a, b) {
     return a + b;
   },
   subtract(a, b) {
     return a - b;
   },
   multiply(a, b) {
     return a  b;
   },
   divide(a, b) {
     return a / b;
   },
 };
 
 // Пример использования
 console.log(Calculator.add(10, 5)); // 15
