    <script>
        let numbers = [];
        for (let i = 0; i < 15; i++) {
            numbers.push(Math.floor(Math.random() * 41) - 10);
        }
        // Вычисление среднего арифметического
        let sum = 0;
        for (let number of numbers) {
            sum += number;
        }
        const average = sum / numbers.length;

        alert("Исходный массив:\n" + numbers.join(", ") + "\n\nСреднее арифметическое: " + average.toFixed(2));
    </script>
