<script>
const strArray = ['JavaScript', 'Python', 'PHP', 'Java', 'C'];
const checkLength = (arr, len) => {
  const newArr = [];
  for (let i = 0; i < arr.length; i++) {
    if (arr[i].length <= len) {
      newArr.push(0);
    } else {
      newArr.push(1);
    }
  }
  return newArr;
};
const lenArray = checkLength(strArray, 3);
alert(lenArray);
